Database Creation 
CREATE DATABASE map_management;
USE map_management;

-- Students
CREATE TABLE students (
    prn VARCHAR(20) PRIMARY KEY,
    first_name VARCHAR(50),
    middle_name VARCHAR(50),
    last_name VARCHAR(50),
    dept VARCHAR(100),
    year INT,
    programme VARCHAR(50),
course_duration INT,
 admission_year VARCHAR(9),
    password VARCHAR(255)
);

-- Coordinators
CREATE TABLE coordinators (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    password VARCHAR(255)
);

-- HoDs
CREATE TABLE hods (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    password VARCHAR(255)
);

-- Admins
CREATE TABLE admins (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    password VARCHAR(255)
);

-- Activities submitted by students
CREATE TABLE activities (
    id INT AUTO_INCREMENT PRIMARY KEY,
    prn VARCHAR(20),
    category CHAR(1), -- A-E
    activity_type VARCHAR(100),
    level VARCHAR(20),
    certificate VARCHAR(255),
    date DATE,
    remarks TEXT,
    status ENUM('Pending','Approved','Rejected') DEFAULT 'Pending',
    points INT DEFAULT 0,
    FOREIGN KEY (prn) REFERENCES students(prn)
);


CREATE TABLE programme_rules (
    id INT AUTO_INCREMENT PRIMARY KEY,
    admission_year VARCHAR(9),   -- e.g., '2025-2026'
    programme VARCHAR(50),
    duration INT,
    technical INT,
    sports_cultural INT,
    community_outreach INT,
    innovation INT,
    leadership INT,
    total_points INT
);

üîπ Step 2: Activities Table
Each category has multiple sub-activities.
We‚Äôll store activity type, required proof, and whether points are fixed or level-based.
CREATE TABLE activities_master (
    id INT AUTO_INCREMENT PRIMARY KEY,
    category_id CHAR(1),
    activity_name VARCHAR(150),
    document_evidence VARCHAR(150),
    points_type ENUM('Fixed','Level'), -- whether fixed points or depends on level
    min_points INT DEFAULT NULL,       -- for fixed or minimum
    max_points INT DEFAULT NULL,       -- for fixed or maximum
    FOREIGN KEY (category_id) REFERENCES categories(id)
);

üîπ Step 3: Activity Levels Table
Some activities (Categories A, B, E) have different points based on level (College ‚Üí International, Dept ‚Üí University).
CREATE TABLE activity_levels (
    id INT AUTO_INCREMENT PRIMARY KEY,
    activity_id INT,
    level VARCHAR(50),    -- College, District, State, National, International, Dept, University
    points INT,
    FOREIGN KEY (activity_id) REFERENCES activities_master(id) ON DELETE CASCADE
);
